* ファンの設定
.bashrcに以下を書き込み
function fan() {
    if [ $1 -ge 0 -a $1 -le 255 ]; then
	sudo sh -c "echo $1 > /sys/devices/pwm-fan/target_pwm"
    else
	echo 'must be 0~255'
    fi
}
